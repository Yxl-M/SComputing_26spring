#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main(int argc, char *argv[])
{
    if (argc != 5)
    {
        printf("Usage: %s N a x y\n", argv[0]);
        printf("Example: %s 10 3 0.1 7.1\n", argv[0]);
        return 1;
    }

    long long N = atoll(argv[1]);
    double a = atof(argv[2]);
    double xval = atof(argv[3]);
    double yval = atof(argv[4]);

    if (N <= 0)
    {
        printf("Error: N must be positive.\n");
        return 1;
    }

    double *x = (double *)malloc(N * sizeof(double));
    double *y = (double *)malloc(N * sizeof(double));
    double *d = (double *)malloc(N * sizeof(double));

    if (x == NULL || y == NULL || d == NULL)
    {
        printf("Memory allocation failed for N = %lld\n", N);
        free(x);
        free(y);
        free(d);
        return 1;
    }

    for (long long i = 0; i < N; i++)
    {
        x[i] = xval;
        y[i] = yval;
    }

    for (long long i = 0; i < N; i++)
    {
        d[i] = a * x[i] + y[i];
    }

    double expected = a * xval + yval;
    double tol = 1e-12;
EOF return 0;rintf("d[%lld] = %.17g\n", i, d[i]);in tolerance %.1e\n", tol);
